```python
class SearchOrchestrator:
    """
    搜索编排器 — 根据问题类型智能调度搜索引擎
    """
    
    STRATEGY = {
        "解题方法":    ["tavily", "exa"],           # 语义搜索优先
        "政策查询":    ["tavily", "brave"],          # 双引擎交叉验证
        "深度研究":    ["gemini_deep_research"],     # 后台深度调研
        "视频推荐":    ["media_search", "tavily"],   # 媒体搜索优先
        "学习方法":    ["xiaohongshu", "tavily"],    # 学生视角优先
        "时事热点":    ["brave", "tavily"],           # 新闻类搜索
        "综合搜索":    ["gongzhonghao_search"],      # 多引擎综合
    }
    
    FALLBACK_CHAIN = ["tavily_node", "tavily_bash", "brave", "exa"]
    
    async def search(self, query: str, intent: str) -> list:
        """
        1. 根据意图选择搜索引擎组合
        2. 并发调用多个引擎
        3. 结果融合、去重、排序
        4. 引擎故障时自动降级到备选链
        """
        engines = self.STRATEGY.get(intent, self.FALLBACK_CHAIN[:2])
        results = await asyncio.gather(*[
            self._call_engine(engine, query) for engine in engines
        ])
        return self._merge_and_rank(results)
```

```python
class ZhongkaoRAGRetriever:
    """
    中考专家 RAG 检索器
    三级检索 + 重排序
    """
    
    def retrieve(self, query: str, subject: str = None, top_k: int = 5):
        # === 第一级：向量语义检索 ===
        vector_results = self.chroma_db.query(
            query_embeddings=self.embed(query),
            n_results=top_k * 3,  # 多取后排序
            where={"subject": subject} if subject else None
        )
        
        # === 第二级：关键词精确匹配 ===
        # 数学公式、化学方程式等需要精确匹配
        keyword_results = self.keyword_search(query)
        
        # === 第三级：知识点关联扩展 ===
        # "将军饮马" → 关联 "对称变换"、"最短路径"
        related_points = self.knowledge_graph.expand(query)
        related_results = self.chroma_db.query(
            query_embeddings=self.embed(related_points),
            n_results=top_k
        )
        
        # === 融合 + 重排序 ===
        all_results = self._merge(vector_results, keyword_results, related_results)
        reranked = self.reranker.rank(query, all_results)
        
        return reranked[:top_k]


# RAG + 搜索引擎协同策略
class KnowledgeRouter:
    """
    知识路由器 — 决定从哪里获取知识
    """
    
    def route(self, query: str, intent: str) -> str:
        # 1. 首先查RAG本地知识库
        rag_results = self.rag.retrieve(query)
        
        if rag_results and rag_results[0].score > 0.85:
            # RAG置信度高，直接使用
            return "rag_only", rag_results
        
        elif rag_results and rag_results[0].score > 0.6:
            # RAG有相关内容，搜索引擎补充
            search_results = self.searcher.search(query, intent)
            return "rag_plus_search", (rag_results, search_results)
        
        else:
            # RAG无相关内容，依赖搜索引擎
            search_results = self.searcher.search(query, intent)
            return "search_only", search_results
```

```javascript
const VENUS_API = 'http://v2.open.venus.oa.com/llmproxy/v1/chat/completions';
const VENUS_KEY = 'YOUR_VENUS_API_TOKEN';

async function streamChat(messages) {
    const response = await fetch(VENUS_API, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${VENUS_KEY}`
        },
        body: JSON.stringify({
            model: 'claude-opus-4-6',
            messages: messages,
            stream: true
            // 可选: reasoning模式用于复杂数学推导
        })
    });
    
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    
    while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        const lines = chunk.split('\n');
        for (const line of lines) {
            if (line.startsWith('data: ') && line !== 'data: [DONE]') {
                const data = JSON.parse(line.slice(6));
                const content = data.choices?.[0]?.delta?.content;
                if (content) yield content;
            }
        }
    }
}
```

```
📐 首先画出这个三角形：
[交互式几何图形 - JSXGraph渲染]
  · A(0,0), B(6,0), C(0,8)
  · 可拖拽查看

💡 关键发现：
AB=6, BC=8, AC=10
检查: 6² + 8² = 36 + 64 = 100 = 10² = AC²
→ 这是一个**直角三角形**！∠A = 90°

📝 解题过程：
$$S = \frac{1}{2} \times AB \times AC_{⊥}$$

$$S = \frac{1}{2} \times 6 \times 8 = 24$$

✅ 答案：三角形面积 = 24
```
